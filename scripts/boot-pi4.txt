setenv autostart yes

# The Raspberry Pi 4's physical SD card slot is driven by the "emmc2"
# controller, while mmc@7e300000 is actually the onboard WiFi SDIO chip (see
# bcm2711-rpi-4-b.dts's &sdhci node, "SDHCI is used to control the SDIO for
# wireless"). That WiFi device is non-removable and always logs a benign
# "Card did not respond to voltage select!" during bootflow scanning. Which
# u-boot mmc index (0 or 1) ends up assigned to emmc2 depends on the node
# order of the devicetree that was loaded (it flips between u-boot's own
# bundled bcm2711-rpi-4-b.dtb and Raspberry Pi's official firmware one), so
# try both instead of hardcoding a single index.
if load mmc 0:1 $kernel_addr_r mini.elf || load mmc 1:1 $kernel_addr_r mini.elf; then
	bootelf $kernel_addr_r $fdt_addr $kernel_addr_r
else
	echo "Unable to read mini.elf"
fi
sleep 10000000
